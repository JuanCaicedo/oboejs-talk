Object Oriented?

                  
oboe('/myapp/people')
  .node('people.*', function(person){
    return new Person(
      person.firstName,
      person.lastName
    );
  })
  .done(function(people){
    // OOP people!
  })

Filtering?

                    
var onlyPeople = [];

oboe('/myapp/people')
  .node('people.*', function(person){
    if(person.name !== 'Roger Rabbit'){
      onlyPeople.push(person);
    }
  })
  .done(function(){
    console.log(onlyPeople) // no R.R :(
  })

Memory?

                    
oboe('/myapp/people')
  .node('{ firstName lastName }', function(person){
    // write to the DOM
    return oboe.drop;
  })
  .done(function( finalJson){
    console.log(finalJson); // {"people": []}
  })
