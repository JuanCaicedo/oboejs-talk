Interact on the fly!

Object Oriented?

              
oboe('/myapp/people')
  .node('people.*', function(person){
    return new Person(person.firstName, person.lastName)
  })
  .done(function(people){
    // OOP people!
  })

Filtering

              
var onlyPeople = [];

oboe('/myapp/people')
  .node('people.*', function(person){

    if(person.name !== 'Duke'){
      onlyPeople.push(person);
    }

  })

  .done(function(){
    console.log(onlyPeople) // no Duke :(
  })

Memory efficient


oboe('/myapp/people')

    // write to the DOM

  })

  })
